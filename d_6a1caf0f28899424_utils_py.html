94def escapeString(val, maxLength=254): 

95 """ 

96 Quotes several characters and removes "\\\\n" and "\\\\0" to prevent XSS injection. 

97 

98 :param val: The value to be escaped. 

99 :type val: str 

100 

101 :param maxLength: Cut-off after maxLength characters. A value of 0 means "unlimited". 

102 :type maxLength: int 

103 

104 :returns: The quoted string. 

105 :rtype: str 

106 """ 

107 val = str(val).strip() \ 

108 .replace("<", "&lt;") \ 

109 .replace(">", "&gt;") \ 

110 .replace("\"", "&quot;") \ 

111 .replace("'", "&#39;") \ 

112 .replace("(", "&#040;") \ 

113 .replace(")", "&#041;") \ 

114 .replace("=", "&#061;") \ 

115 .replace("\n", "") \ 

116 .replace("\0", "") 

117 

118 if maxLength: 

119 return val[0:maxLength] 

120 

121 return val

143def downloadUrlFor(folder: str, fileName: str, derived: bool = False, 

144 expires: Union[timedelta, None] = timedelta(hours=1), 

145 downloadFileName: Optional[str] = None) -> str: 

146 """ 

147 Utility function that creates a signed download-url for the given folder/filename combination 

148 

149 :param folder: The GCS-Folder (= the download-key) for that file 

150 :param fileName: The name of that file. Either the original filename as uploaded or the name of a dervived file 

151 :param derived: True, if it points to a derived file, False if it points to the original uploaded file 

152 :param expires: None if the file is supposed to be public (which causes it to be cached on the google ede 

153 caches), otherwise a timedelta of how long that link should be valid 

154 :param downloadName: If set, we'll force to browser to download this blob with the given filename 

155 :return: THe signed download-url relative to the current domain (eg /download/...) 

156 """ 

157 if derived: 

158 filePath = "%s/derived/%s" % (folder, fileName) 

159 else: 

160 filePath = "%s/source/%s" % (folder, fileName) 

161 if downloadFileName: 

162 downloadFileName = sanitizeFileName(downloadFileName) 

163 else: 

164 downloadFileName = "" 

165 expires = ((datetime.now() + expires).strftime("%Y%m%d%H%M") if expires else 0) 

166 sigStr = "%s\0%s\0%s" % (filePath, expires, downloadFileName) 

167 sigStr = urlsafe_b64encode(sigStr.encode("UTF-8")) 

168 resstr = hmacSign(sigStr) 

169 return "/file/download/%s?sig=%s" % (sigStr.decode("ASCII"), resstr)

171def srcSetFor(fileObj: dict, expires: Optional[int], width: Optional[int] = None, height: Optional[int] = None) -> str: 

172 """ 

173 Generates a string suitable for use as the srcset tag in html. This functionality provides the browser 

174 with a list of images in different sizes and allows it to choose the smallest file that will fill it's viewport 

175 without upscaling. 

176 :param fileObj: The file-bone (or if multiple=True a single value from it) to generate the srcset for 

177 :param expires: None if the file is supposed to be public (which causes it to be cached on the google ede 

178 caches), otherwise it's lifetime in seconds 

179 :param width: A list of widths that should be included in the srcset. If a given width is not available, it will 

180 be skipped. 

181 :param height: A list of heights that should be included in the srcset. If a given height is not available, 

182 it will be skipped. 

183 :return: The srctag generated or an empty string if a invalid file object was supplied 

184 """ 

185 if not width and not height: 

186 logging.error("Neither width or height supplied to srcSetFor") 

187 return "" 

188 if "dlkey" not in fileObj and "dest" in fileObj: 

189 fileObj = fileObj["dest"] 

190 if expires: 

191 expires = timedelta(minutes=expires) 

192 if not isinstance(fileObj, (SkeletonInstance, dict)) or not "dlkey" in fileObj or "derived" not in fileObj: 

193 logging.error("Invalid fileObj supplied to srcSetFor") 

194 return "" 

195 if not isinstance(fileObj["derived"], dict): 

196 return "" 

197 resList = [] 

198 for fileName, derivate in fileObj["derived"]["files"].items(): 

199 customData = derivate.get("customData", {}) 

200 if width and customData.get("width") in width: 

201 resList.append("%s %sw" % (downloadUrlFor(fileObj["dlkey"], fileName, True, expires), customData["width"])) 

202 if height and customData.get("height") in height: 

203 resList.append("%s %sh" % (downloadUrlFor(fileObj["dlkey"], fileName, True, expires), customData["height"])) 

204 return ", ".join(resList)
