17def pbkdf2(password, salt, iterations=1001, keylen=42): 

18 """ 

19 An implementation of PBKDF2 (http://wikipedia.org/wiki/PBKDF2) 

20 

21 Mostly based on the implementation of 

22 https://github.com/mitsuhiko/python-pbkdf2/blob/master/pbkdf2.py 

23 

24 :copyright: (c) Copyright 2011 by Armin Ronacher. 

25 :license: BSD, see LICENSE for more details. 

26 """ 

27 _pack_int = Struct('>I').pack 

28 if isinstance(password, str): 

29 password = password.encode("UTF-8") 

30 if isinstance(salt, str): 

31 salt = salt.encode("UTF-8") 

32 mac = hmac.new(password, None, sha256) 

33 

34 def _pseudorandom(x, mac=mac): 

35 h = mac.copy() 

36 h.update(x) 

37 return h.digest() 

38 

39 buf = [] 

40 for block in range(1, -(-keylen // mac.digest_size) + 1): 

41 rv = u = _pseudorandom(salt + _pack_int(block)) 

42 for i in range(iterations - 1): 

43 u = _pseudorandom((''.join(map(chr, u))).encode("LATIN-1")) 

44 rv = starmap(xor, zip(rv, u)) 

45 buf.extend(rv) 

46 return codecs.encode(''.join(map(chr, buf))[:keylen].encode("LATIN-1"), 'hex_codec')

49class passwordBone(stringBone): 

50 """ 

51 A bone holding passwords. 

52 This is always empty if read from the database. 

53 If its saved, its ignored if its values is still empty. 

54 If its value is not empty, its hashed (with salt) and only the resulting hash 

55 will be written to the database

91 if not name in data:
